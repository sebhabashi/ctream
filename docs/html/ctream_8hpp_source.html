431 class BasicArena
432 {
433 public:
434  BasicArena() noexcept
435  {
436  }
437 
438  ~BasicArena()
439  {
440  // Destroy all the objects
441  std::lock_guard<std::mutex> objLk{m_objectsMx};
442  for (auto& ptr : m_objects)
443  ptr.delFct(ptr.ptr);
444 
445  // Free the memory
446  std::lock_guard<std::mutex> lk{m_pagesListMx};
447  for (auto& page : m_pages)
448  {
449  std::lock_guard<std::mutex> pageLk{page.mx};
450  freePage(page);
451  }
452  }
453 
454  void* allocate(size_t size) noexcept
455  {
456  // This is the only way to ensure page is protected exactly the right
457  // amount of time
458  // It cannot interlock because no thread has to wait for
459  // m_pagesListMx to unlock page.mx
460 
461  // Access/create a page with enough space
462  m_pagesListMx.lock();
463  auto& page = firstAvailablePage(size);
464  page.mx.lock();
465  m_pagesListMx.unlock();
466 
467  // Allocate on the page
468  void* data = (void*) (size_t(page.data) + page.cursor);
469  page.cursor += size;
470  page.mx.unlock();
471 
472  return data;
473  }

475  template<typename U, typename... Args>
476  U* construct(Args&&... args)
477  {
478  void* data = allocate(sizeof(U));
479  auto* obj = new (data) U(std::forward<Args>(args)...);
480 
481  {
482  std::lock_guard<std::mutex> lk{m_objectsMx};
483  m_objects.emplace_back();
484  auto& ptr = m_objects.back();
485  ptr.ptr = data;
486  ptr.delFct = voidDeleterUseWithCaution<U>;
487  }
488 
489  return obj;
490  }

512  static void initPage(size_t minSize, Page& out) noexcept
513  {
514  out.size = (minSize > fine_tuning::PAGE_SIZE)
515  ? minSize
516  : fine_tuning::PAGE_SIZE;
517  out.cursor = 0;
518  out.data = malloc(out.size);
519  }

522  {

526  inline static long occupiedSizeOnPage(const Page& p) noexcept
527  {
528  return long(p.cursor);
529  }
530 
531  inline static long availableSizeOnPage(const Page& p) noexcept
532  {
533  return long(p.size) - occupiedSizeOnPage(p);
534  }
535 
536  Page& firstAvailablePage(size_t size) noexcept
537  {
538  for (auto& p : m_pages)
539  {
540  if (availableSizeOnPage(p) > size)
541  return p;
542  }
543  m_pages.emplace_back();
544  auto& newPage = m_pages.back();
545  initPage(size, newPage);
546  return newPage;

742  template<typename A, typename R = A>
743  R collect(const collectors::Collector<T, A, R>& collector) const
744  {
745  constexpr size_t MULTITHREAD_MIN_SIZE =
746  internal::fine_tuning::MULTITHREAD_MIN_SIZE;
747  constexpr size_t THREADS_PER_CORE =
748  internal::fine_tuning::THREADS_PER_CORE;
749 
750  const size_t nThreads = std::min(
751  1 + (m_containerSize) / MULTITHREAD_MIN_SIZE,
752  size_t(THREADS_PER_CORE * std::thread::hardware_concurrency()));
753 
754  if (nThreads < 2)
755  {
756  // If only 1 thread is used, do directly in current thread
757  A a = collector.supply();
758  for (size_t i = 0; i < m_containerSize; ++i)
759  {
760  // Compute the i'th item
761  const T* item = computeItem(i);
762  if (item)
763  collector.accumulate(a, *item);
764  }
765  return collector.finish(a);
766  }
767  else
768  {
769  // Initialize empty containers
770  std::vector<A> chunks;
771  std::vector<std::thread> threads;
772  chunks.reserve(nThreads);
773  threads.reserve(nThreads);
774 
775  // Accumulate values in separate chunks
776  size_t first = 0;
777  for (size_t i = 0; i < nThreads; ++i)
778  {
779  chunks.emplace_back(collector.supply());
780  auto& chunk = chunks.back();
781 
782  const size_t last = (i == nThreads - 1)
783  ? (m_containerSize)
784  : (first + m_containerSize / nThreads);
785 
786  threads.emplace_back(
787  [this, first, last, &chunk, &collector] ()
788  {
789  for (size_t j = first; j < last; ++j)
790  {
791  // Collect item (if not filtered out)
792  const T* item = computeItem(j);
793  if (item)
794  collector.accumulate(chunk, *item);
795  }
796  });
797 
798  // Next thread picks up where this thread left
799  first = last;
800  }
801 
802  // Wait for all chunks to be ready
803  for (auto& thread : threads)
804  if (thread.joinable())
805  thread.join();
806 
807  // Combine all chunks
808  A a = collector.supply();
809  for (auto& chunk : chunks)
810  collector.combine(a, chunk);
811 
812  return collector.finish(a);
813  }
814  }
